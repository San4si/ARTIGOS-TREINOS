Envia a usuario e a senha em toda requisição

import { Request, Response, NextFunction} from 'express';
import {User} from 'model/user'

export const Auth = {
    private: async (req: Request, res: Response, next: NextFunction) => {
    
        let sucess = false;
        //fazer verificação de auth
        if(req.headers.authorization){
            let hash: string = req.headers.authorization.substring(6);
            let decoded: string = Buffer.from(hash, 'base64').toString();// converte um base64 em string
            let data: string[] = decoded.split(':');//separa o email da senha

            if(data.length === 2){//verificando se data tem dois itens
                let hashUser = await User.findOne({
                    where: {
                        email: data[0],
                        password: data[1]
                    }
                });
                if(hashUser){
                    sucess = true;
                }
            }
        }   

        if(sucess){
            next();
        }else {
            res.status(403); //Not Authorized
            res.json({ error: 'Não autorizado' });
        }
    }
}
